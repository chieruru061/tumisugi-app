実装イメージ

``` C# 
using System.ComponentModel.DataAnnotations;

namespace TumisugiApp.Models.ViewModels
{
    /// <summary>
    /// お積みものViewModel. お積みもの1recordを表現する。
    /// </summary>
    public class TumiItemViewModel
    {
        [Required]
        public int TumId { get; set; }
        
        [Required]
        [StringLength(100)]
        public required string Title { get; set; }

        [Required]
        public required string GanreName { get; set; }

        [Required]
        public required string PlatformName { get; set; }

        [Required]
        public required string StatusName { get; set; }
    }

    /// <summary>
    /// 積みリスト検索条件ViewModel（リクエストパラメータ用）。
    /// </summary>
    public class TumiListRequestViewModel
    {
        public string? SearchTitle { get; set; }  // タイトル検索
        public string? SelectedGenre { get; set; }  // ジャンル選択
        public string? SelectedPlatform { get; set; }  // プラットフォーム選択
        public string? SelectedStatus { get; set; }  // ステータス選択
        // 必要に応じてページングやソート条件を追加
    }

    /// <summary>
    /// 積みリスト表示用ViewModel（リスト全体 + 検索条件）。
    /// </summary>
    public class TumiListViewModel
    {
        public IEnumerable<TumiItemViewModel> Items { get; set; } = new List<TumiItemViewModel>();
        public TumiListRequestViewModel Request { get; set; } = new TumiListRequestViewModel();
    }
}
```

2. Service
``` C#
// TumiListService.cs
using TumisugiApp.Models.ViewModels;
using TumisugiApp.Repositories;

namespace TumisugiApp.Services
{
    public class TumiListService
    {
        private readonly ITumiItemRepository _repository;

        public TumiListService(ITumiItemRepository repository)
        {
            _repository = repository;
        }

        public async Task<IEnumerable<TumiItemViewModel>> GetAllViewModelsAsync()
        {
            var items = await _repository.GetAllAsync();
            return items.Select(i => new TumiItemViewModel
            {
                TumId = i.TumId,
                Title = i.Title,
                GanreName = i.GanreName,
                PlatformName = i.PlatformName,
                StatusName = i.StatusName
            });
        }
    }
}
```

3. Controller
``` C#
// TumiListController.cs
using TumisugiApp.Services;

namespace TumisugiApp.Controllers
{
    public class TumiListController : Controller
    {
        private readonly TumiListService _service;

        public TumiListController(TumiListService service)
        {
            _service = service;
        }

        public async Task<IActionResult> Index()
        {
            var viewModels = await _service.GetAllViewModelsAsync();
            return View(viewModels);
        }
    }
}
```

4. DI登録
